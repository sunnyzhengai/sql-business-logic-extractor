readmit_risk
  Classified as:
  - 'High' when Diagnosis name/description matches pattern '%heart failure%' and Length of stay in days (calculated) is greater than 7
  - 'Medium' when Diagnosis name/description matches pattern '%pneumonia%'
  - 'Medium' when Time patient departed the Emergency Department is not null
  - 'Low' otherwise
  Source: CLARITY_EDG (Diagnosis reference table (ICD-10, descriptions)), PAT_ENC_HSP (Hospital patient encounters (inpatient, ED, observation))
  Technical: CASE WHEN clarity_edg.dx_name LIKE '%heart failure%' AND pat_enc_hsp.los_days > 7 THEN 'High' WHEN clarity_edg.dx_name LIKE '%pneumonia%' THEN 'Medium' WHEN NOT pat_enc_hsp.ed_departure_time IS NULL THEN 'Medium' ELSE 'Low' END

visit_type
  Classified as:
  - 'Inpatient' when 1
  - 'Outpatient' when 2
  - 'Emergency' when 3
  - 'Other' otherwise
  Source: PAT_ENC_HSP (Hospital patient encounters (inpatient, ED, observation))
  Technical: CASE pat_enc_hsp.adt_pat_class_c WHEN 1 THEN 'Inpatient' WHEN 2 THEN 'Outpatient' WHEN 3 THEN 'Emergency' ELSE 'Other' END
