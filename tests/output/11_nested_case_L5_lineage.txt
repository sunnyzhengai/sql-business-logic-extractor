PAT_ENC_CSN_ID (passthrough)
  Resolved: PAT_ENC_HSP.PAT_ENC_CSN_ID
  Base columns: PAT_ENC_HSP.PAT_ENC_CSN_ID
  Base tables: PAT_ENC_HSP

readmit_risk (case)
  Resolved: CASE WHEN dx.DX_NAME LIKE '%heart failure%' AND e.LOS_DAYS > 7 THEN 'High' WHEN dx.DX_NAME LIKE '%pneumonia%' THEN 'Medium' WHEN NOT e.ED_DEPARTURE_TIME IS NULL THEN 'Medium' ELSE 'Low' END
  Base columns: CLARITY_EDG.DX_NAME, PAT_ENC_HSP.LOS_DAYS, PAT_ENC_HSP.ED_DEPARTURE_TIME
  Base tables: CLARITY_EDG, PAT_ENC_HSP

visit_type (case)
  Resolved: CASE e.ADT_PAT_CLASS_C WHEN 1 THEN 'Inpatient' WHEN 2 THEN 'Outpatient' WHEN 3 THEN 'Emergency' ELSE 'Other' END
  Base columns: PAT_ENC_HSP.ADT_PAT_CLASS_C
  Base tables: PAT_ENC_HSP
