DEPARTMENT_NAME (passthrough)
  Resolved: CLARITY_DEP.DEPARTMENT_NAME
  Base columns: CLARITY_DEP.DEPARTMENT_NAME
  Base tables: CLARITY_DEP
  Filters:
    - NOT e.HOSP_DISCH_TIME IS NULL

encounter_count (aggregate)
  Resolved: COUNT(DISTINCT e.PAT_ENC_CSN_ID)
  Base columns: PAT_ENC_HSP.PAT_ENC_CSN_ID
  Base tables: PAT_ENC_HSP
  Filters:
    - NOT e.HOSP_DISCH_TIME IS NULL

avg_los (aggregate)
  Resolved: AVG(DATEDIFF(DAY, e.HOSP_ADMSN_TIME, e.HOSP_DISCH_TIME))
  Base columns: PAT_ENC_HSP.DAY, PAT_ENC_HSP.HOSP_ADMSN_TIME, PAT_ENC_HSP.HOSP_DISCH_TIME
  Base tables: PAT_ENC_HSP
  Filters:
    - NOT e.HOSP_DISCH_TIME IS NULL
