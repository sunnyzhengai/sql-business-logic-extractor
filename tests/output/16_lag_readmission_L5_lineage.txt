prev_discharge (window)
  Resolved: LAG(HOSP_DISCH_TIME) OVER (PARTITION BY PAT_ID ORDER BY HOSP_ADMSN_TIME)
  Base columns: PAT_ENC_HSP.PAT_ID, PAT_ENC_HSP.HOSP_DISCH_TIME, PAT_ENC_HSP.HOSP_ADMSN_TIME
  Base tables: PAT_ENC_HSP
  Filters:
    - NOT HOSP_DISCH_TIME IS NULL

days_since_last_discharge (window)
  Resolved: DATEDIFF(DAY, LAG(HOSP_DISCH_TIME) OVER (PARTITION BY PAT_ID ORDER BY HOSP_ADMSN_TIME), HOSP_ADMSN_TIME)
  Base columns: PAT_ENC_HSP.DAY, PAT_ENC_HSP.PAT_ID, PAT_ENC_HSP.HOSP_DISCH_TIME, PAT_ENC_HSP.HOSP_ADMSN_TIME
  Base tables: PAT_ENC_HSP
  Filters:
    - NOT HOSP_DISCH_TIME IS NULL
