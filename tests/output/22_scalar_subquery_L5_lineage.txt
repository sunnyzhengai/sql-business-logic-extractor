PAT_ENC_CSN_ID (passthrough)
  Resolved: PAT_ENC_HSP.PAT_ENC_CSN_ID
  Base columns: PAT_ENC_HSP.PAT_ENC_CSN_ID
  Base tables: PAT_ENC_HSP
  Filters:
    - fm.PAT_ENC_CSN_ID = e.PAT_ENC_CSN_ID
    - fm.FLO_MEAS_ID = '5'

last_bp_time (subquery)
  Resolved: (SELECT MAX(fm.RECORDED_TIME) FROM IP_FLWSHT_MEAS AS fm WHERE fm.PAT_ENC_CSN_ID = e.PAT_ENC_CSN_ID AND fm.FLO_MEAS_ID = '5')
  Base columns: IP_FLWSHT_MEAS.RECORDED_TIME, IP_FLWSHT_MEAS.PAT_ENC_CSN_ID, PAT_ENC_HSP.PAT_ENC_CSN_ID, IP_FLWSHT_MEAS.FLO_MEAS_ID
  Base tables: IP_FLWSHT_MEAS, PAT_ENC_HSP
  Filters:
    - fm.PAT_ENC_CSN_ID = e.PAT_ENC_CSN_ID
    - fm.FLO_MEAS_ID = '5'
