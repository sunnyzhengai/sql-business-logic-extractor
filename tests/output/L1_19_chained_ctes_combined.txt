FORMATTED SQL:
----------------------------------------
WITH discharges AS (
  SELECT
    PAT_ID,
    PAT_ENC_CSN_ID,
    HOSP_DISCH_TIME,
    DISCH_DISPOSITION_C
  FROM PAT_ENC_HSP
  WHERE
    NOT HOSP_DISCH_TIME IS NULL AND ADT_PAT_CLASS_C = 1
), readmissions AS (
  SELECT
    d1.PAT_ID,
    d1.PAT_ENC_CSN_ID AS index_csn,
    d2.PAT_ENC_CSN_ID AS readmit_csn,
    DATEDIFF(DAY, d1.HOSP_DISCH_TIME, d2.HOSP_DISCH_TIME) AS days_to_readmit
  FROM discharges AS d1
  JOIN discharges AS d2
    ON d1.PAT_ID = d2.PAT_ID
    AND d2.HOSP_DISCH_TIME > d1.HOSP_DISCH_TIME
    AND DATEDIFF(DAY, d1.HOSP_DISCH_TIME, d2.HOSP_DISCH_TIME) <= 30
)
SELECT
  r.PAT_ID,
  p.PAT_MRN_ID,
  r.index_csn,
  r.readmit_csn,
  r.days_to_readmit
FROM readmissions AS r
JOIN PATIENT AS p
  ON r.PAT_ID = p.PAT_ID
ORDER BY
  r.days_to_readmit

BUSINESS DEFINITIONS (Plain English):
----------------------------------------
  filter:where:HOSP_DISCH_TIME
    Filter: only include records where hosp_disch_time is not null
    Technical: NOT hosp_disch_time IS NULL

  filter:where:ADT_PAT_CLASS_C
    Filter: only include records where adt_pat_class_c equals 1
    Technical: adt_pat_class_c = 1

  filter:where:HOSP_DISCH_TIME
    Filter: only include records where hosp_disch_time is not null
    Technical: NOT hosp_disch_time IS NULL

  filter:where:ADT_PAT_CLASS_C
    Filter: only include records where adt_pat_class_c equals 1
    Technical: adt_pat_class_c = 1

  days_to_readmit
    The number of days between discharges.hosp_disch_time and discharges.hosp_disch_time
    Source: discharges
    Technical: DATEDIFF(day, discharges.hosp_disch_time, discharges.hosp_disch_time)
