FORMATTED SQL:
----------------------------------------
SELECT
  dept_stats.DEPARTMENT_NAME,
  dept_stats.avg_los,
  dept_stats.encounter_count
FROM (
  SELECT
    dep.DEPARTMENT_NAME,
    AVG(DATEDIFF(DAY, e.HOSP_ADMSN_TIME, e.HOSP_DISCH_TIME)) AS avg_los,
    COUNT(*) AS encounter_count
  FROM PAT_ENC_HSP AS e
  JOIN CLARITY_DEP AS dep
    ON e.DEPARTMENT_ID = dep.DEPARTMENT_ID
  WHERE
    NOT e.HOSP_DISCH_TIME IS NULL
  GROUP BY
    dep.DEPARTMENT_NAME
) AS dept_stats
WHERE
  dept_stats.avg_los > 5

    Source: dept_stats

  avg_los
    Average of DAY, Hospital admission date and time, Hospital discharge date and time
    Source: PAT_ENC_HSP (Hospital patient encounters (inpatient, ED, observation))
    Condition: Hospital discharge date and time is not null
    Technical: AVG(DATEDIFF(day, pat_enc_hsp.hosp_admsn_time, pat_enc_hsp.hosp_disch_time))

  encounter_count
    Count of records
    Technical: COUNT(*)

  filter:where:PAT_ENC_HSP.HOSP_DISCH_TIME
    Filter: only include records where Hospital discharge date and time is not null
    Source: PAT_ENC_HSP (Hospital patient encounters (inpatient, ED, observation))
    Technical: NOT pat_enc_hsp.hosp_disch_time IS NULL
