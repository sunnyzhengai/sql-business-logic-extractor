FORMATTED SQL:
----------------------------------------
WITH enc AS (
  SELECT
    e.PAT_ID,
    e.PAT_ENC_CSN_ID
  FROM PAT_ENC_HSP AS e
  WHERE
    e.ADT_PAT_CLASS_C = 1
)
SELECT
  PAT_ID,
  PAT_ENC_CSN_ID
FROM enc

----------------------------------------
  filter:where:PAT_ENC_HSP.ADT_PAT_CLASS_C

    Source: PAT_ENC_HSP (Hospital patient encounters (inpatient, ED, observation))
